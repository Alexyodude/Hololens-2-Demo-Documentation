    Toggle Drone

    This turns on and off the drone.

    • Namespaces:

      • System and System.Collections are used for basic functionalities and collections.
      • UnityEngine provides access to Unity-specific classes and methods needed for game development.

      • Variables:

      • public GameObject explosion: Prefab for the explosion effect. It's used to show a visual effect when a drone is destroyed.

      • public GameObject DronePrefab: Prefab for the drone that will be instantiated in the game.
      • public GameObject user: The GameObject representing the user or player, used to determine the position relative to which drones are placed and moved.
      • private List<GameObject> Drones: Keeps track of all active drones. This is a private list to manage the state of drones internally.
      • public int numDrone: Number of drones to create when OnDrone is called.
      • Boolean isDrone: Tracks whether drones are currently active or not.

      • Update Method:

      • Runs every frame. If isDrone is true, it iterates through the Drones list.

      • Moves drones towards the user if they are too far away or below a certain height. The movement is done using Vector3.MoveTowards.

      • OnDrone Method:

      • Spawns a number of drones at random positions around the user. The drones are instantiated and added to the Drones list.

      • Sets isDrone to true to indicate that drones are active.

      • OffDrone Method:

      • Destroys all active drones and clears the Drones list.

      • Sets isDrone to false to indicate that drones are no longer active.

      • droneDestroy Method:

      • Handles the destruction of an individual drone.

      • Instantiates an explosion effect at the drone’s position and destroys it after 2 seconds.
      • Optionally moves the drone's position close to the user (which might be used for some specific gameplay or visual effect).

    using System; // Import the System namespace for basic data types and functions
using System.Collections.Generic; // Import the System.Collections.Generic namespace for generic collection classes like List
using UnityEngine; // Import the UnityEngine namespace to access Unity-specific classes and methods

public class ToggleDrone : MonoBehaviour
{
    // Public variables for easy assignment in the Unity Inspector
    public GameObject explosion; // Prefab for the explosion effect when a drone is destroyed
    public GameObject DronePrefab; // Prefab for the drone that will be instantiated
    public GameObject user; // Reference to the user GameObject that the drones will interact with
    private List<GameObject> Drones = new List<GameObject>(); // List to keep track of active drones
    public int numDrone = 10; // Number of drones to spawn when activated
    Boolean isDrone = false; // Boolean to track if drones are active or not

    private void Update()
    {
        // This method is called once per frame
        if (isDrone) // Check if drones are active
        {
            foreach (GameObject drone in Drones) // Iterate over each drone in the list
            {
                if (drone != null) // Ensure the drone is not null
                {
                    // Check if the drone is too far from the user or below a certain height
                    if (Vector3.Distance(drone.transform.position, user.transform.position) > 4 || drone.transform.position.y < -0.2)
                    {
                        // Move the drone towards the user
                        drone.transform.position = Vector3.MoveTowards(drone.transform.position, user.transform.position, 0.1f);
                    }
                }
            }
        }
    }

    public void OnDrone()
    {
        // Method to activate and spawn drones
        for (int i = 0; i < numDrone; i++) // Loop to create a number of drones
        {
            // Generate a random position within a circular area around the user
            Vector3 random = UnityEngine.Random.insideUnitCircle * 2;
            Vector3 pos = new Vector3(user.transform.position.x + random.x, user.transform.position.y + random.y, user.transform.position.z + random.y);
            // Instantiate a drone at the calculated position
            GameObject Drone = Instantiate(DronePrefab, pos, Quaternion.identity, gameObject.transform);
            // Add the instantiated drone to the list
            Drones.Add(Drone);
        }
        isDrone = true; // Set the flag to indicate that drones are active
    }

    public void OffDrone()
    {
        // Method to deactivate and destroy all drones
        foreach (GameObject drone in Drones) // Iterate over each drone in the list
        { 
            Destroy(drone); // Destroy the drone GameObject
        }
        Drones.Clear(); // Clear the list of drones
        isDrone = false; // Set the flag to indicate that drones are inactive
    }

    public void droneDestroy(GameObject drone)
    {
        // Method to handle drone destruction with explosion effect
        // Instantiate explosion effect at the drone's position and destroy it after 2 seconds
        Destroy(Instantiate(explosion, drone.transform.position, Quaternion.identity), 2f);
        // Optionally reset the drone's position to a location close to the user
        drone.transform.position = user.transform.position + user.transform.forward * 0.5f;
    }
}
